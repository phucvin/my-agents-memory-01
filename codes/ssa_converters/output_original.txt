entry:
    x = Const(1)
    y = Const(10)
    Jump(if_check)
if_check:
    cond = <(x, 5)
    Branch(cond, then, else)
then:
    x = +(x, 1)
    Jump(loop_header)
else:
    x = *(x, 2)
    Jump(loop_header)
loop_header:
    loop_cond = >(y, 0)
    Branch(loop_cond, loop_body, exit)
loop_body:
    y = -(y, 1)
    x = +(x, y)
    Jump(loop_header)
exit:
    Print(x)
    Return(0)
